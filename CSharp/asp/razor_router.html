                                                          # 路由参数

                                                          @page "/router/{text}" 可选路由 @page "/router/{text?}"

                                                          # 路由约束

                                                          类型约束示例是否可选
                                                          bool{active:bool}true，FALSE否
                                                          datetime{dob:datetime}2016-12-31，2016-12-31 7:32pm是
                                                          decimal{price:decimal}49.99，-1,000.01是
                                                          double{weight:double}1.234，-1,001.01e8是
                                                          float{weight:float}1.234，-1,001.01e8是
                                                          guid{id:guid}CD2C1638-1638-72D5-1638-DEADBEEF1638否
                                                          int{id:int}123456789，-123456789是
                                                          long{ticks:long}123456789，-123456789是

                                                          # 导航

                                                          @page "/navigate"
@using Microsoft.Extensions.Logging 
@implements IDisposable
@inject ILogger<Navigate> Logger
@inject NavigationManager Navigation

<h1>Navigate in component code example</h1>

<button class="btn btn-primary" @onclick="NavigateToCounterComponent">
    Navigate to the Counter component
</button>

@code {
    private void NavigateToCounterComponent()
    {
        Navigation.NavigateTo("counter");
    }

    protected override void OnInitialized()
    {
        Navigation.LocationChanged += HandleLocationChanged;
    }

    private void HandleLocationChanged(object? sender, LocationChangedEventArgs e)
    {
        Logger.LogInformation("URL of new location: {Location}", e.Location);
    }

    public void Dispose()
    {
        Navigation.LocationChanged -= HandleLocationChanged;
    }
}
